Add plugins to your OpenCode configuration

opencode.json
{
  "$schema": "https://opencode.ai/config.json",
  "plugin": [
    "opencode-dynamic-context-pruning",
    "opencode-morph-plugin",
    "opencode-model-fallback",
    "claude-code-safety-net",
    "opencode-notify"
  ]
}
